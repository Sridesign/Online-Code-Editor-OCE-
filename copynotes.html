Exercise 3

Simple Counter

<h2 id="count"
    style="text-align:center;font-size:3rem;">
  0
</h2>
<div style="text-align:center">
  <button onclick="change(-1)">−</button>
  <button onclick="change(1)">+</button>
</div>

<script>
  let n = 0;
  function change(v) {
    n += v;
    document.getElementById("count").innerText = n;
  }
</script>
